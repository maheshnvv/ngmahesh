🚀 Features

🎯 Simple Integration

Convert any div to a map with just a directive: ngOsmMap

📍 Flexible Pins

Support for coordinates, addresses, cities, states, and postal codes

🎨 Customization

Custom pin colors, HTML content, and custom icons

Highlight areas with custom borders, fill colors, and opacity

Built-in OpenStreetMap Nominatim API integration

⚡ Dynamic Updates

Add, remove, and update pins dynamically at runtime

📦 Installation

Install the library and dependencies: 
npm install ng-osm-map leaflet @types/leaflet

Add Leaflet CSS to your angular.json:

"styles": [
  "node_modules/leaflet/dist/leaflet.css",
  "src/styles.scss"
]

Ensure HttpClient is provided (for geocoding):

import { provideHttpClient } from '@angular/common/http';

export const appConfig: ApplicationConfig = {
  providers: [
    provideHttpClient(),
    // ... other providers
  ]
};

💻 Basic Usage

Using the Directive: 
import { Component } from '@angular/core';
import { NgOsmMapDirective, PinObject, LocationObject } from 'ng-osm-map';

@Component({
  selector: 'app-map',
  standalone: true,
  imports: [NgOsmMapDirective],
  template: `
    <div
      ngOsmMap
      [pins]="pins"
      [zoomInto]="centerLocation"
      [highlightAreas]="areas"
      style="height: 400px; width: 100%;">
    </div>
  `
})
export class MapComponent {
  pins: PinObject[] = [
    {
      location: { latitude: 40.7128, longitude: -74.0060 },
      color: '#ff0000',
      title: 'New York City',
      content: '<h3>NYC</h3><p>The Big Apple</p>'
    },
    {
      location: {
        address: '1600 Amphitheatre Parkway',
        city: 'Mountain View',
        state: 'CA'
      },
      color: '#00ff00',
      title: 'Google HQ'
    }
  ];

  centerLocation: LocationObject = {
    latitude: 40.7128,
    longitude: -74.0060
  };
}

📚 API Reference

Core Interfaces

LocationObject

interface LocationObject {
  latitude?: number;
  longitude?: number;
  address?: string;
  city?: string;
  state?: string;
  country?: string;
  zipCode?: string;
}

PinObject

interface PinObject {
  location: LocationObject;
  color?: string;
  content?: string;
  title?: string;
  icon?: string | PinIcon;
}

HighlightArea

interface HighlightArea {
  boundary: LocationObject[];
  borderColor?: string;
  borderWidth?: number;
  fillColor?: string;
  fillOpacity?: number;
  title?: string;
}
